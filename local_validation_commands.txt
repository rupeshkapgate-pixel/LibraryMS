python -m pip install -r services\lending-service\requirements.txt

################################################################################
# 12. PYTHON LINT - SERVICE-WISE OPTIONAL COMMANDS
################################################################################

python -m ruff check services/lending-service/app/ services/lending-service/tests/ --ignore E501,E402 --exclude "*/proto_generated/*"
